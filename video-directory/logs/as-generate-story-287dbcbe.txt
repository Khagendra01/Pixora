      {daisies.map(({ delay, startX, startY }, index) => {

              top: floatY,

              height: \"auto\",
              opacity: interpolate(loopProgress, [0, 0.1, 0.9, 1], [0, 1, 1, 0]),
            }}
          />
        );
      })}

      <div

          position: \"absolute\",
          left: \"50%\",

          textAlign: \"center\",
